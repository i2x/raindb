n_rr=nrow(rf)

ye_rain=rf[n_rr,1]
n.yr=ye_rain-sy_rain+1
year=sy_rain:ye_rain
mth=5			# starting month of season

r.m=matrix(r.m[][r.m[,2]<(mon+1)],ncol=3)
ss.rain[i,1]=sum(r.m[,3],na.rm=FALSE)
}
